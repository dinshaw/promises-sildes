Initialize

  def initialize
    @state = :pending
    @value = nil
    begin
      yield method(:fulfill), method(:reject)
    rescue Exception => e
      reject(e)
    end
  end

Any

  def self.any(*promises)
    Promise.new do |fulfill, reject|
      count = promises.size
      on_error = ->(*) do
        count -= 1
        reject.call if count == 0
      end
      promises.each do |promise|
        promise.then(fulfill, on_error)
      end
    end
  end
